# 📑 Спецификация семантического манифеста AURA (v3.1)

### Расширение протокола: Распределенная семантическая память, контракты рантайма и межсистемный обмен ДНК

Этот документ определяет строгий стандарт YAML для блоков распределенных семантических метаданных (`manifest.md`), используемых **AURA NexusCompiler** и сетевым протоколом **AI Net**. Версия 3.1 переводит манифест из статического описания архитектуры в исполняемый семантический контракт (Executable Semantic Contract) для сквозной автоматизации, безопасного рефакторинга силами ИИ и динамического взаимодействия цифровых организмов.

## 🛠️ Полная схема метаданных (Максимальный уровень: Level 3)


```yaml
# ==============================================================================
# AURA SEMANTIC MANIFEST METADATA SCHEMA v3.1
# ==============================================================================

# --- БЛОК 1: ФУНДАМЕНТ ИДЕНТИФИКАЦИИ И КООРДИНАЦИИ ---
identity:
  node: "/Context/Assembly"         # Целевой относительный или глобальный URI узла системы.
  type: "memory_layer"               # Архетип узла. Допустимо: [root_manifest, system_core, memory_layer, gateway_router, execution_agent, data_sink]
  subsystem: "Context Assembly"     # Человеко- и машиночитаемое имя функционального блока.
  version: "2026.3.1"                # Версия на основе SemVer или эпохи, отслеживающая состояние этого узла.
  stability_layer: "stable"          # Операционная зрелость. Допустимо: [experimental, active, stable, legacy]
  criticality: "mission_critical"    # Системная важность. Допустимо: [low, medium, high, mission_critical]

# --- БЛОК 2: АРХИТЕКТУРНЫЕ СВЯЗИ СИСТЕМЫ ---
architecture_relations:
  db_relations:
    - table: "ai_sessions"
      mode: "write/read"            # Вектор доступа. Допустимо: [read_only, write_only, write/read]
      purpose: "Логирует и синхронизирует интерактивные матрицы контекста пользователя и токены выполнения."
    - table: "company_registry"
      mode: "read_only"
      purpose: "Валидирует заголовки входящего трафика на соответствие зарегистрированным корпоративным криптографическим ключам."
  
  external_connections:
    - system: "Qdrant Vector DB"
      protocol: "gRPC"              # Протокол транспортного уровня: [gRPC, HTTPS/REST, WebSockets, MQTT, Redis-RESP]
      endpoint: "vector-cluster:6334"
      purpose: "Извлечение глубоких слоев семантических знаний и ассоциативных весов."

# --- БЛОК 3: МАТРИЦЫ ЖЕСТКИХ И МЯГКИХ ЗАВИСИМОСТЕЙ ---
dependencies:
  upstream:                         # Что требуется этому узлу для успешного запуска (Входящие зависимости)
    - node: "/BDGSite"
      type: "hard"                  # Жесткая связь. Если этот узел дает сбой или отсутствует, текущий узел умирает.
      purpose: "Требуются предварительно смашрутизированные, валидированные пакеты транзакций с извлеченными ключами компании."
  downstream:                       # Узлы, которые напрямую зависят от этого узла (Исходящие зависимости)
    - node: "/Worker/Consultant"
      type: "soft"                  # Мягкая связь. Может работать с ограниченной функциональностью в случае отсутствия.
      purpose: "Потребляет скомпилированные массивы контекста для выполнения лингвистических конвейеров агентов."

# --- БЛОК 4: ИНТЕРФЕЙСНЫЕ КОНТРАКТЫ ДАННЫХ (Новое в v3.1) ---
interface_contracts:
  input:
    - name: "compiled_task"
      format: "json"
      required: true
      source: "/BDGSite"
      purpose: "Получение валидированных пакетов задач после пре-роутинга и извлечения company_key."
    - name: "user_context"
      format: "json"
      required: false
      source: "/Session"
      purpose: "Опциональный контекст пользователя/сессии для обогащения собираемой матрицы."
  output:
    - name: "context_package"
      format: "json"
      consumer: "/Worker/Consultant"
      purpose: "Предоставление собранного семантического контекста для исполнительных ИИ-агентов."

# --- БЛОК 5: ПРОФИЛИ РЕСУРСОВ И ОГРАНИЧЕНИЯ ---
resource_profiles:
  load_expectations:
    target_rps: "250"               # Ожидаемый или протестированный базовый уровень запросов в секунду.
    peak_rps: "1000"                # Максимальная пропускная способность до срабатывания лимитера (rate-limiting).
  resilience:
    timeout_ms: "450"               # Максимальное время обработки до генерации исключения выполнения.
    retry_policy: "exponential_backoff_3" # Стратегия повторных запросов при ошибке.
    offline_capability: false       # True, если узел может продолжать выполнение логики без внешних сетевых подключений.

# --- БЛОК 6: ЭВОЛЮЦИОННЫЕ И МИГРАЦИОННЫЕ МАРКЕРЫ ---
evolution:
  potential: "high"                 # Вероятность алгоритмических изменений. Допустимо: [low, medium, high]
  migration_path: "v2_to_v3_schema" # Указатель на скрипт слоя трансляции для изменений базы данных или состояний.
  domain_ontology: "cognitive_layer" # Тег классификации онтологии знаний для автономного индексирования.

# --- БЛОК 7: ПОЛИТИКИ БЕЗОПАСНОСТИ И АУДИТА ---
security_policies:
  access_level: "private"          # Уровень доступности. Допустимо: [public, internal, private, isolated]
  encryption: "TLS_1.3"             # Стандарт шифрования каналов, используемый для всего исходящего трафика узла.
  audit_vector: "syslog_structured" # Целевой обработчик для логирования телеметрии и проверок доступа.

# --- БЛОК 8: ЧУВСТВИТЕЛЬНОСТЬ И КЛАССИФИКАЦИЯ ДАННЫХ (Новое в v3.1) ---
data_sensitivity:
  contains_personal_data: true       # Наличие персональных данных пользователей.
  contains_financial_data: false     # Наличие платежных, банковских или транзакционных данных.
  contains_credentials: false        # Наличие сырых паролей, токенов, приватных ключей доступа.
  contains_trade_secrets: true       # Наличие коммерческой тайны или закрытых алгоритмов ядра.
  contains_user_generated_content: true # Наличие контента, генерируемого пользователями (UGC).
  retention_policy: "30_days"        # Срок хранения данных в рантайме этого узла до очистки.
  anonymization_required: true       # Требуется ли деперсонализация перед передачей во внешние контуры.
  public_exposure_allowed: false     # Разрешена ли выдача структуры/данных узла через публичный /semantic-ping.

# --- БЛОК 9: ТЕСТОВЫЕ СОГЛАШЕНИЯ И РЕГРЕССИЯ (Новое в v3.1) ---
test_contracts:
  required_tests:
    - "context_assembly_integrity"
    - "company_key_validation"
    - "qdrant_connection_health"
  regression_scope:                  # Зона влияния: что может сломаться в соседних модулях при изменении текущего
    - "/BDGSite"
    - "/Context/Assembly"
    - "/Worker/Consultant"
  failure_policy: "block_deployment" # Действие при падении тестов. Допустимо: [log_warning, degrade_service, block_deployment]
  ai_review_required: true           # Требуется ли обязательный аудит кода ИИ-супервизором перед коммитом.

# --- БЛОК 10: НАБЛЮДАЕМОСТЬ И ТЕЛЕМЕТРИЯ (Новое в v3.1) ---
observability:
  healthcheck: "/health/context-assembly" # Эндпоинт проверки операционной жизнеспособности.
  metrics:
    - "rps"
    - "latency_ms"
    - "error_rate"
    - "qdrant_query_time"
  alert_thresholds:
    latency_ms: 450
    error_rate_percent: 3
    timeout_count_per_minute: 10
  telemetry_target: "prometheus"     # Целевой агрегатор метрик.
  log_level: "structured_info"       # Глубина логирования.

```

## 📘 Подробная спецификация полей и полезность для ИИ

### Блок 1: Фундамент идентификации и координации

* **`identity.node`** (String) — Абсолютная структурная точка привязки компонента.
* *Полезность для ИИ:* Задает точные координаты узла в графе системы. Позволяет распределенным ИИ обмениваться картами топологии и понимать физическое расположение логических блоков.


* **`identity.type`** (Enum) — Архетип узла (`root_manifest`, `system_core`, `memory_layer` и т.д.).
* *Полезность для ИИ:* Мгновенно переключает когнитивные эвристики LLM. Для `gateway_router` ИИ применит паттерны сетевой безопасности, для `execution_agent` — алгоритмы операционного ветвления.


* **`identity.stability_layer`** & **`identity.criticality`** (Enum) — Стабильность и критичность компонента для бизнеса.
* *Полезность для ИИ:* Моделирует карту рисков. При генерации кода или рефакторинге модуля со статусом `mission_critical` ИИ автоматически выстраивает максимальный контур верификации и изоляции изменений.



### Блок 2: Архитектурные связи системы

* **`architecture_relations.db_relations`** (Array) — Карта таблиц и баз данных, с которыми взаимодействует узел.
* *Полезность для ИИ:* Полностью исключает разрушение схем данных. Проверяя флаг `mode` (`read_only` против `write/read`), ИИ-разработчик контролирует права доступа и исключает несанкционированные инъекции записи в критические таблицы.


* **`architecture_relations.external_connections`** (Array) — Инфраструктурные сетевые стыки модуля (протоколы, эндпоинты).
* *Полезность для ИИ:* Обеспечивает автоматическое связывание API (Automated API Binding). ИИ понимает транспортный уровень (`gRPC`, `REST`) и может самостоятельно конфигурировать сетевые коннекторы «из коробки».



### Блок 3: Матрицы жестких и мягких зависимостей

* **`dependencies.upstream`** & **`dependencies.downstream`** (Array) — Прямая декларация входящих и исходящих зависимостей.
* *Полезность для ИИ:* Позволяет строить динамические архитектурные графы. Понимая разницу между `hard` (падение родительского узла уничтожает текущий) и `soft` (узел продолжит жить в режиме деградации функций), ИИ-оркестратор корректно управляет отказоустойчивостью системы.



### Блок 4: Интерфейсные контракты данных *(Новое в v3.1)*

* **`interface_contracts.input`** & **`interface_contracts.output`** (Array) — Спецификация структуры, формата и обязательности прохождения пакетов данных.
* *Полезность для ИИ:* Фундамент для **автономной генерации кода**. ИИ-агенту больше не нужно угадывать типы аргументов и структуру объектов. Контракт жестко описывает, какие JSON-схемы на входе требуются узлу и какой контент он обязуется выдать на выходе.



### Блок 5: Профили ресурсов и ограничения

* **`resource_profiles.load_expectations`** & **`resource_profiles.resilience`** (Map) — Лимиты производительности (RPS), таймауты и правила восстановления.
* *Полезность для ИИ:* Задает рамки написания алгоритмов. При проектировании внутренних функций ИИ использует значение `timeout_ms` для жесткой конфигурации cURL или gRPC запросов прямо внутри исполняемого кода, гарантируя предсказуемость рантайма.



### Блок 6: Эволюционные и миграционные маркеры

* **`evolution`** (Map) — Потенциал изменчивости, пути миграции данных и теги корпоративного Графа Знаний (`domain_ontology`).
* *Полезность для ИИ:* Позволяет ИИ-архитекторам контекстно индексировать систему. Узлы автоматически связываются с общим семантическим ядром бизнеса, помогая ИИ определять онтологическую роль каждого файла.



### Блок 7: Политики безопасности и аудита

* **`security_policies`** (Map) — Уровни изоляции и шифрования каналов.
* *Полезность для ИИ:* Автоматический аудит безопасности. Гарантирует, что ИИ никогда не скомпилирует метод, передающий данные по незащищенному протоколу, если манифест требует `TLS_1.3`.



### Блок 8: Чувствительность и классификация данных *(Новое в v3.1)*

* **`data_sensitivity`** (Map) — Типизация и правила обработки конфиденциальных данных (PII, финансы, коммерческая тайна).
* *Полезность для ИИ:* **Главный Guardrail безопасности.** При обработке запроса по протоколу `AI Net` или обработке внешнего `/semantic-ping`, локальный ИИ сверяется с этим блоком. Если `public_exposure_allowed: false` или `anonymization_required: true`, ИИ превентивно вырезает, маскирует или хэширует данные узла, полностью предотвращая утечки.



### Блок 9: Тестовые соглашения и регрессия *(Новое в v3.1)*

* **`test_contracts`** (Map) — Набор обязательных тестов, зона потенциального поражения соседних модулей (`regression_scope`) и реакция на сбои.
* *Полезность для ИИ:* Позволяет ИИ проводить безопасный рефакторинг. Перед деплоем измененного кода ИИ анализирует `regression_scope` и превентивно тестирует всю цепочку зависимых модулей. Если срабатывает `block_deployment`, поврежденный код изолируется автоматически.



### Блок 10: Наблюдаемость и телеметрия *(Новое в v3.1)*

* **`observability`** (Map) — Точки проверки здоровья (`healthcheck`), ключевые метрики рантайма и критические пороги алертов.
* *Полезность для ИИ:* Основа концепции **самозалечивания систем (Self-Healing)**. ИИ-супервизор инфраструктуры непрерывно сравнивает метрики рантайма со значениями `alert_thresholds`. В случае аномалии (например, задержка > 450мс), ИИ локализует проблему, перенаправляет трафик или оптимизирует кэш без участия DevOps-инженера.



## 📊 Рекомендуемые уровни детализации манифестов (Manifest Levels)

Чтобы оптимизировать контекстное окно ИИ-агентов и избавить разработчиков от избыточного описания простых модулей, спецификация v3.1 разделяет манифесты на три уровня применения:

### Level 1: Минимальный манифест (Minimal)

*Рекомендуется для изолированных хелперов, утилит, статических конфигураций или низкорисковых папок со статикой (css, картинки).*

* **Включает блоки:** `identity` (базовый), `architecture_relations` (пустые массивы), `dependencies` (пустые массивы), `security_policies` (базовый).

### Level 2: Операционный манифест (Operational)

*Рекомендуется для стандартных функциональных модулей, контроллеров, моделей и активных компонентов продакшена.*

* **Включает блоки:** `identity`, `architecture_relations` (с указанием БД и связей), `dependencies` (родословная узла), `resource_profiles`, `security_policies`.

### Level 3: Полный семантический манифест рантайма (Full Semantic Runtime)

*Обязателен для критически важных узлов, участников сети AI Net, распределенных слоев памяти (Memory Layers), семантических шлюзов авторизации (Gateway Routers) и управляющих оркестраторов.*

* **Включает блоки:** Все 10 блоков в максимальной детализации (`identity` до уровня `mission_critical`, `interface_contracts`, `data_sensitivity`, `test_contracts`, `observability`).
